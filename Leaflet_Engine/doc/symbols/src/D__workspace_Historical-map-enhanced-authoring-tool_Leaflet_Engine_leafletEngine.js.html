370 /**
371  *Create a new Layer from a valid GeoJSON Object
372  * @param {<a href="http://geojson.org/geojson-spec.html">GeoJSON</a>} geoJSON_object The GeoJSON object to read
373  * @returns {Layer} The layer created
374  */
375 Layer.createLayerFromGeoJSON = function(geoJSON_object){
376 	
377 	var layer = Layer.createLayer(null, null);
378 	
379 	//if geojson object is a collection of features (=> if the geojson object is a layer) and if properties is defined
380 	if(geoJSON_object.features && geoJSON_object.properties){
381 		if(geoJSON_object.properties.name)layer.setName(geoJSON_object.properties.name);
382 		if(geoJSON_object.properties.description)layer.setDescription(geoJSON_object.properties.description);
383 	}
384 	
385 	var llayer = L.geoJSON(geoJSON_object, {onEachFeature : function(feature, layer){
386 		
387 			var type;
388 			if(feature.geometry == 'Point'){type = Element.geometry.POINT;}	
389 			if(feature.geometry == 'MultiPoint'){type = Element.geometry.MULTIPOINT;}
390 			if(feature.geometry == 'LineString'){type = Element.geometry.LINE;}
391 			if(feature.geometry == 'MultiLineString'){type = Element.geometry.MULTILINE;}
392 			if(feature.geometry == 'Polygon'){type = Element.geometry.POLYGON;}
393 			if(feature.geometry == 'MultiPolygon'){type = Element.geometry.MULTIPOLYGON;}
394 			
395 			var element = Element.createElementFromLeafletLayer(feature.properties.name ? feature.properties.name : null, feature.properties.description ? feature.properties.description : null, type, layer);
396 			layer.addElement(element);
397 		}
398 	});
399 	
400 	//TODO : test what happens if the geojson object is invalid or unrecognized
401 	return layer;
402 	
403 }

489 	 */
490 	saveAsGeoJSON : function(){
491 		
492 		var geojson;
493 		geojson.type = "Feature";
494 		geojson.properties = this.getProperties();
495 		geojson.geometry.type = this.getType();
496 		
497 		var coordinates = [];
498 		
499 		if(this.getType == Element.geometry.POINT){
500 			coordinates.push(this._lelement.getLatLng().lat);
501 			coordinates.push(this._lelement.getLatLng().lng);
502 		}
503 		if(this.getType == Element.geometry.MULTIPOINT){
504 			this._lelement.forEachLayer(function(layer){
505 				var point = Util.latLngToArray(layer.getLatLng());
506 				coordinates.push(point);
507 			});
508 		}
509 		if(this.getType == Element.geometry.LINE){
510 			var coords = this._lelement.getLatLngs();
511 			for(i=0;i<coords.length;i++){
512 				var point = Util.latLngToArray(coords[i].getLatLng());
513 				coordinates.push(point);
514 			}
515 		}
516 		if(this.getType == Element.geometry.MULTILINE){
517 			this._lelement.forEachLayer(function(layer){
518 				var linecoords = [];
519 				var line = layer.getLatLngs();
520 				for(i = 0;i<line.length;i++){
521 					var point = Util.latLngToArray(line[i].getLatLng());
522 					linecoords.push(point);
523 				}
524 				coordinates.push(linecoords);
525 			});
526 		}
527 		if(this.getType == Element.geometry.POLYGON){
528 			var coords = this._lelement.getLatLngs();
529 			var temp = [];
530 			for(i=0;i<coords.length;i++){
531 				var point = Util.latLngToArray(coords[i].getLatLng());
532 				temp.push(point);
533 			}
534 			coordinates.push(temp);
535 		}
536 		if(this.getType == Element.geometry.MULTIPOLYGON){
537 			this._lelement.forEachLayer(function(layer){
538 				var polycoords = [];
539 				var temp = [];
540 				var poly = layer.getLatLngs();
541 				for(i = 0;i<poly.length;i++){
542 					var point = Util.latLngToArray(line[i].getLatLng());
543 					temp.push(point);
544 				}
545 				polycoords.push(temp);
546 				coordinates.push(polycoords);
547 			});
548 		}
549 		
550 		
551 		
552 		geojson.geometry.coordinates = coordinates;
553 		
554 		return geojson;

657  *@param {<a href="http://leafletjs.com/reference.html#polyline-options">L.Polyline_options</a>} leaflet_polygon_options Options for Polygon Element
658  */
659 Element.createPolygon = function(name, desc, latslngs, leaflet_polygon_options){
660 	var elem = Element._createAbstractElement(name, desc);
661 	if(latslngs.length == 2){
662 		elem._lelement = L.Rectangle(L.LatLngBounds(latslngs), leaflet_polygon_options);
663 	}
664 	else{
665 		elem._lelement = L.Polygon(latslngs, leaflet_polygon_options);
666 	}
667 	elem._type = Element.geometry.POLYGON;
668 	return elem;
669 };

671 /**
672  *Create a new MultiPolygon Element with a name, a description and some options with the specified coordinates (lat long coordinates)
673  *@param {String} name the name of the Element
674  *@param {String} desc description of the Element
675  *@param {<a href="http://leafletjs.com/reference.html#latlng">L.LatLng[][]</a>} latlng Positions of the Element on the map (latitude - longitude)
676  *@param {<a href="http://leafletjs.com/reference.html#polyline-options">L.Polyline_options</a>} leaflet_polygon_options Options for Polygon Element
677  */
678 Element.createMultiPolygon = function(name, desc, latslngs, leaflet_polygon_options){
679 	var elem = Element._createAbstractElement(name, desc);
680 	elem._lelement = L.MultiPolygon(latslngs, leaflet_polygon_options);
681 	elem._type = Element.geometry.MULTIPOLYGON;
682 	return elem;
683 }
