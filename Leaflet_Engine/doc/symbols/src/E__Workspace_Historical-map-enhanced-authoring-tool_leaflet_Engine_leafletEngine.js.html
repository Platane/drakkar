141 /**
142  *Create a map with a name, a description, the div id where to put it and some options
143  * @see <a href="http://leafletjs.com/reference.html#map-options">leaflet map options</a>
144  * @param {String} name The name of the map
145  * @param {String} desc A description of the map
146  * @param {String} div_id The div id of the html element where to put this map
147  * @param {<a href="http://leafletjs.com/reference.html#map-options">Map_Options</a>} leaflet_map_options Options for the map
148  * @returns {Map} A map with a name, a description and some options (when this function is 
149  * 	called the map will be automatically drawn on the specified html element) 
150  */
151 Map.createMap = function( name, desc, div_id, leaflet_map_options ){
152 	var m = new Map();
153 	m._lmap = new L.Map(div_id, leaflet_map_options);
154 	m._name = name;
155 	m._description = desc;
156 	return m;
157 }

310  */
311 Layer.createLayerFromGeoJSON = function( name, desc, geoJSON_object){
312 	
313 	var layer = Layer.createLayer(name, desc);
314 	var llayer = L.geoJSON(geoJSON_object, {onEachFeature : _onEachFeature, style : _geoJSONStyle});
315 	
316 	llayer.onEachLayer(function ( layer ){
317 		layer.addElement( layer );
318 	});
319 	//TODO : test what happens if the geojson object is invalid or unrecognized
320 	return layer;
321 	
322 }
